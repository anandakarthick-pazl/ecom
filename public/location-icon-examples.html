PHP/Laravel Usage Examples

Code Examples
1. Using the IconClass Helper
// Get all location icons
$locationIcons = \App\Helpers\IconClass::getLocationIcons();

// Get default location icon
$defaultIcon = \App\Helpers\IconClass::getDefaultLocationIcon();

// Render location icon HTML
$iconHtml = \App\Helpers\IconClass::renderLocationIcon('fas fa-store', '#28a745', '50px');

// Check if icon is valid
$isValid = \App\Helpers\IconClass::isValidIcon('fas fa-map-marker-alt');

// Get icon name
$iconName = \App\Helpers\IconClass::getIconName('fas fa-store'); // Returns "Store"
2. In Blade Templates
{{-- Use the location icon selector component --}}
@include('components.location-icon-selector', [
    'fieldName' => 'location_icon',
    'label' => 'Choose Location Icon',
    'value' => $location->icon ?? 'fas fa-map-marker-alt',
    'required' => true,
    'previewColor' => '#dc3545'
])

{{-- Display location icon --}}
<div class="location-icon" style="background-color: {{ $location->color }}">
    <i class="{{ $location->icon_class }} text-white"></i>
</div>
3. Controller Usage
// In your controller
use App\Helpers\IconClass;

class LocationController extends Controller
{
    public function create()
    {
        $locationIcons = IconClass::getLocationIcons();
        $predefinedPlatforms = IconClass::getPredefinedPlatforms();
        
        return view('admin.locations.create', compact('locationIcons', 'predefinedPlatforms'));
    }
    
    public function store(Request $request)
    {
        $request->validate([
            'icon_class' => ['required', 'string', function ($attribute, $value, $fail) {
                if (!IconClass::isValidIcon($value)) {
                    $fail('The selected icon is not valid.');
                }
            }],
        ]);
        
        // Store location with icon
        Location::create($request->validated());
    }
}
